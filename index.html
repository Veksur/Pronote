const int ledPins[5] = {2, 3, 4, 5, 6};

void setup() {
  Serial.begin(9600);
  for (int i = 0; i < 5; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
}

void loop() {
  if (Serial.available() > 0) {
    int count = Serial.read() - '0';
    for (int i = 0; i < 5; i++) {
      digitalWrite(ledPins[i], i < count ? HIGH : LOW);
    }
  }
}
